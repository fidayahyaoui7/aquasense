📝 Exemples de requêtes

-- Voir tous les utilisateurs
SELECT * FROM users;

-- Supprimer un utilisateur (remplacez 1 par l'ID)
DELETE FROM users WHERE id = 1;

-- Modifier le prénom d'un utilisateur
UPDATE users SET prenom = 'Nouveau' WHERE id = 1;

-- Voir les relevés d'un utilisateur spécifique
SELECT * FROM readings WHERE user_id = 1;

-- Supprimer tous les relevés d'un utilisateur
DELETE FROM readings WHERE user_id = 1;

-- Supprimer toutes les alertes d'un utilisateur
DELETE FROM alerts WHERE user_id = 1;

-- Compter les utilisateurs par type de bâtiment
SELECT building_type, COUNT(*) as total FROM users GROUP BY building_type;

-- Voir les alertes non résolues
SELECT * FROM alerts WHERE resolved = 0;
